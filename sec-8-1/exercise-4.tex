\Que{
    Find the exact length of the curve
    $ f(x) = \frac{x^3}{3} + \frac{1}{4x}, 1 \le x \le 2 $\\
    }
\Ans{
    $ L = \int_{1}^{2} {\sqrt{1+[f^{\prime}(x)]^2}} dx $\\
    
    $ f^{\prime}(x) = x^2-\frac{1}{4x^2} $\\
    
    $ [f^{\prime}(x)]^2 = [x^2-\frac{1}{4x^2}]^2 $\\
    
    $ = x^4 + \frac{1}{16x^4} - \frac{1}{2} $\\
    
    $ L = \int_{1}^{2} {\sqrt{1+x^4 + \frac{1}{16x^4} - \frac{1}{2}}} dx $\\
    
    $ = \int_{1}^{2} {\sqrt{x^4 + \frac{1}{16x^4} + \frac{1}{2}}} dx $\\

    $ = \int_{1}^{2} {\sqrt{\frac{16x^8 + 1 + 8x^4 }{16x^4}}} dx $\\
    
    $ = \int_{1}^{2} {\frac{4x^4 + 1}{4x^2}} dx 
    = \int_{1}^{2} {x^2 + \frac{1}{4x^2}} dx
    = \left[\frac{x^3}{3} - \frac{1}{4x} \right]_{1}^{2} $\\

    $ = \left[\frac{8}{3} - \frac{1}{8} - (\frac{1}{3} - \frac{1}{4}) \right]
    = \frac{7}{3} + \frac{1}{8} $\\
    
    $= \frac{59}{24} $\\
}
