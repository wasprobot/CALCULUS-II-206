\Que{
    Find the exact length of the curve
    $ x = \frac{\sqrt{y}(y-3)}{3}, 1 \le y \le 4 $\\
    }
\Ans{
    $ L = \int_{1}^{4} {\sqrt{1+[f^{\prime}(y)]^2}} dy $\\
    
    $ f^{\prime}(y) = \frac{1}{3}\frac{d}{dy}(y^{\frac{3}{2}} - 3y^{\frac{1}{2}})
    = \frac{1}{3} (\frac{3}{2}y^{\frac{1}{2}} - \frac{3}{2}y^{\frac{-1}{2}})
    = \frac{1}{2} (y^{\frac{1}{2}} - y^{\frac{-1}{2}}) $\\
    
    $ [f^{\prime}(y)]^2
    = \frac{1}{4} (y + \frac{1}{y} - 2)
    = \frac{(y-1)^2}{4y} $\\

    $ L = \int_{1}^{4} {\sqrt{1 + \frac{(y-1)^2}{4y}}} dy
    = \int_{1}^{4} {\frac{y+1}{2\sqrt{y}}} dy
    = \frac{1}{2} \int_{1}^{4} {(y^{\frac{1}{2}} + y^{-\frac{1}{2}})} dy $\\

    $ = \frac{1}{2} \left[
        \frac{2}{3}y^{\frac{3}{2}} 
        + 2y^{\frac{1}{2}}
    \right]_{1}^{4} $\\
    
    $ = \frac{1}{2} \left[
        \frac{2}{3}4^{\frac{3}{2}} + 2.4^{\frac{1}{2}}
        -(\frac{2}{3} + 2)
    \right]
    = \frac{1}{2}(\frac{14}{3} + 2)
    = \frac{10}{3} $\\
}
