\Que{
    Find the exact length of the curve
    $ x = \frac{y^4}{8} + \frac{1}{4y^2}, 1 \le x \le 2 $\\
    }
\Ans{
    $ L = \int_{1}^{2} {\sqrt{1+[f^{\prime}(y)]^2}} dy $\\
    
    $ f^{\prime}(y) = \frac{y^3}{2} - \frac{1}{2y^3} $\\
    
    $ L = \int_{1}^{2} {\sqrt{1+[\frac{y^3}{2} - \frac{1}{2y^3}]^2}} dy $\\
    
    $ = \int_{1}^{2} {\sqrt{1+[\frac{y^6-1}{2y^3}]^2}} dy $\\
    
    $ = \int_{1}^{2} {\sqrt{1+\frac{y^{12}+1-2y^6}{4y^6}}} dy
    = \int_{1}^{2} {\sqrt{\frac{y^{12}+1+2y^6}{4y^6}}} dy $\\

    $ = \int_{1}^{2} {\frac{y^6+1}{2y^3}} dy
    = \int_{1}^{2} {\frac{y^3}{2} + \frac{1}{2y^3}} dy $\\
    
    $ = \left[ \frac{y^4}{8} - \frac{1}{4y^2} \right]_{1}^{2}
    = \left[ \frac{16}{8} - \frac{1}{16} - (\frac{1}{8} - \frac{1}{4}) \right] $\\
    
    $ = \frac{15}{8} + \frac{3}{16}
    = \frac{33}{16} $\\
}
