\Que{
    Find the exact length of the curve
    $ y = \frac{x^2}{4} - \frac{\ln{x}}{2}, 1 \le x \le 2 $\\
    }
\Ans{
    $ L = \int_{1}^{2} {\sqrt{1+[f^{\prime}(x)]^2}} dx $\\
    
    $ f^{\prime}(x) = \frac{x}{2} - \frac{1}{2x} $\\

    $ L = \int_{1}^{2} {\sqrt{1+[\frac{x}{2} - \frac{1}{2x}]^2}} dx
    = \int_{1}^{2} {\sqrt{1+[\frac{x^2-1}{2x}]^2}} dx
    = \int_{1}^{2} {\sqrt{\frac{(x^2+1)^2}{4x^2}}} dx $\\
    
    $ = \int_{1}^{2} {\frac{x^2+1}{2x}} dx
    = \int_{1}^{2} {(\frac{x}{2} + \frac{1}{2x})} dx $\\
    
    $ = \left[
        \frac{x^2}{4} + \frac{\ln{x}}{2}
    \right]_{1}^{2} $\\

    $ = \left[
        \frac{2^2}{4} + \frac{\ln{2}}{2}
        - (\frac{1^2}{4} + \frac{\ln{1}}{2})
    \right] $\\

    $ = \frac{\ln{2}}{2} + \frac{3}{4} $\\
}
