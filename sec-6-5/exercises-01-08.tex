\Que{
    Find $f_{ave}$, where $f(x)=3x^2+4x$, on $[-1,3]$\\
    }
\Ans{
    $ \int{f(x)dx} = \int{(3x^2+4x)dx} $\\
    
    $ = x^3 + 2x^2 + C $\\
    
    $ \implies f_{ave} = \frac{\left[x^3 + 2x^2\right]_{-1}^{3}}{3+1} $\\
    
    $ = \frac{(3^3+2.3^2) - ((-1)^3+2(-1)^2)}{4} $\\
    
    $ = \frac{(27+18) - (-1+2)}{4} $\\
    
    $ = \frac{45 - 1}{4} = 11 $\\
}
    
\Que{
    Find $f_{ave}$, where $f(x)=5\cos(x) $, on $[-\pi/2,\pi/2]$\\
    }
\Ans{
    $ \int{f(x)dx} = \int{5\cos(x)dx} $\\

    $ = 5\sin(x) + C $\\

    $ \implies f_{ave} = \frac{\left[5\sin(x)\right]_{\pi/2}^{-\pi/2}}{-\pi/2-\pi/2} $\\

    $ = \frac{(-5)-(5)}{-\pi} $\\

    $ = \frac{10}{\pi} $\\
}

\Que{
    Find $f_{ave}$, where $f(t)=\frac{t}{\sqrt{3+t^2}} $, on $ [1,3] $\\
    }
\Ans{
    $ \int{f(t)}dt = \int{\frac{t}{\sqrt{3+t^2}}}dt $\\

    $ = \int{\frac{t}{\sqrt{3+t^2}}}dt = \frac{1}{2}\int{\frac{1}{\sqrt{u}}}du $, for $ u=3+t^2 $\\

    $ = \sqrt{u} + C = \sqrt{(3+t^2)} + C $\\

    $ \implies f_{ave} = \frac{\left[ \sqrt{(3+t^2)} \right]_{1}^{3}}{3-1} $\\
    
    $ = \frac{(\sqrt{3+9})-(\sqrt{3+1})}{2} = \frac{\sqrt{12}-2}{2} = \sqrt{3}-1 $\\
}

\Que{
    Find $f_{ave}$, where $f(t)=e^{\sin(t)}\cos(t) $, on $ [0,\pi/2] $\\
    }
\Ans{
    $ \int{f(t)}dt = \int{e^{\sin(t)}\cos(t)}dt $\\

    $ = \int{e^{u}}du $, for $ u=\sin(t) $\\

    $ = e^{u} = e^{\sin(t)} +C $\\

    $ \implies f_{ave} = \frac{\left[ e^{\sin(t)} \right]_{0}^{\pi/2}}{\pi/2-0} $\\

    $ = \frac{e^1-e^0}{\pi/2} = \frac{e-1}{\pi/2} = \frac{2(e-1)}{\pi} $\\
}
\Que{
    Find $f_{ave}$, where $f(x) = 6\cos^4(x)\sin(x) $, on $ [0,\pi] $\\
    }
\Ans{
    $ \int{f(x)}dx = \int{6\cos^4(x)\sin(x)}dx $\\

    $ = -6\int{u^4}du $, for $ u = \cos(x) $\\

    $ = -\frac{6}{5}u^5 + C = -\frac{6}{5}\cos^5(x) + C $\\

    $ \implies f_{ave} = -\frac{6}{5} \frac{\left[ \cos^5(x) \right]_{0}^{\pi}}{\pi-0} $\\
    
    $ = -\frac{6}{5} \frac{\cos^5(\pi) - \cos^5(0)}{\pi} $\\

    $ = -\frac{6}{5} \frac{-1 - 1}{\pi} = \frac{12}{5\pi} $\\
}
