\Que{
    $ \int{\frac{x^3}{\sqrt{x^2+4}}}dx $\\
    }
\Ans{
    $ \int{\frac{x^3}{\sqrt{x^2+4}}} dx = \int{\frac{x^3}{\sqrt{x^2+2^2}}}dx $\\

    Let $ x=2\tan(\theta) $\\

    $\therefore \sqrt{x^2+2^2} 
    = \sqrt{2^2\tan^2(\theta)+2^2} 
    = 2\sqrt{\tan^2(\theta)+1} 
    = 2\sec(\theta) $\\

    $ \therefore \int{\frac{x^3}{\sqrt{x^2+4}}} dx 
    = \int{\frac{8\tan^3(\theta)}{2\sec(\theta)}} dx
    = 4\int{\frac{\tan^3(\theta)}{\sec(\theta)}} dx $\\

    $ = 4\int{\frac{\sin^3(\theta)}{\cos^2(\theta)}} dx
    = 4\int{\frac{\sin^3(\theta)}{\cos^2(\theta)}} dx $\\


}
