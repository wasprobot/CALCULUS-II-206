\Que{
    Convergent/Divergent?
    $ \int_{2}^{\infty}{e^{-2p}} dp $\\
    }
\Ans{
    $ = \lim_{t \to \infty} \int_{2}^{t}{e^{-2p}} dp $\\
    
    $\int{e^{-2p}} dp 
    = -\frac{1}{2} \int{e^{u}} du $, for $ u=-2p $ and $ dp=-\frac{1}{2}du 
    = -\frac{1}{2} e^{u} + C $\\

    $ = -\frac{1}{2} e^{-2p} + C $\\
    
    $ \therefore \left[-\frac{1}{2} e^{-2p} + C \right]_{2}^{t} $\\

    $ = -\frac{1}{2} \left[e^{-2t} - e^{-4} \right] $\\
    
    $ = -\frac{1}{2} \left[\frac{1}{e^{2t}} - \frac{1}{e^{4}} \right] $\\
    
    $ \therefore -\frac{1}{2} \lim_{t \to \infty} \left[\frac{1}{e^{2t}} - \frac{1}{e^{4}} \right] $\\
    
    $ = -\frac{1}{2} (\lim_{t \to \infty} \frac{1}{e^{2t}} - \frac{1}{e^{4}}) $\\
    
    $ = -\frac{1}{2} (0 - \frac{1}{e^{4}}) $\\
    
    $ = \frac{1}{2e^4} $\\
}
